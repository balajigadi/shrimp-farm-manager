rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isOwnerByFarmId() {
      return signedIn() && resource.data.farmId == request.auth.uid;
    }

    function isOwnerByRequestFarmId() {
      return signedIn() && request.resource.data.farmId == request.auth.uid;
    }

    // Per-user settings doc: userSettings/{uid}
    match /userSettings/{uid} {
      allow read, write: if signedIn() && request.auth.uid == uid;
    }

    // Farm-scoped collections
    match /ponds/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    match /waterLogs/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    match /feedLogs/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    match /growthSamples/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    match /mortalityLogs/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    match /expenses/{docId} {
      allow create: if isOwnerByRequestFarmId();
      allow read, update, delete: if isOwnerByFarmId();
    }

    // Deny everything else by default
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
